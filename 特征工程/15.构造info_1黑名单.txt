--支持格式化输入 ${yyyyMMdd +/- 3d}
--使用PAI提供的UDF(内测功能): https://lark.alipay.com/ai/pai/udf
select t1.info_1,t1.avg_fr_info_1
from
(
select info_1,count(is_fraud)as opco,sum(is_fraud)as opsum ,avg(is_fraud) as avg_fr_info_1
from ${t1}
group by info_1
 )t1
where t1.avg_fr_info_1==1.0 and t1.opco>=2.0
